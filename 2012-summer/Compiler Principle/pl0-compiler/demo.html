PL/0 Compiler Demo

Sample Code

PROGRAM add
    VAR a, b;
    BEGIN
        INPUT a; INPUT b;
        OUTPUT a + b
    END

PROGRAM max
    VAR a, b;
    BEGIN
        INPUT a; INPUT b;
        IF b > a THEN
            a := b;
        OUTPUT a
    END

PROGRAM gcd
    VAR a, b, t;
    BEGIN
        INPUT a; INPUT b;
        WHILE b <> 0 DO
        BEGIN
            t := b;
            b := a % b;
            a := t
        END;
        OUTPUT a
    END
